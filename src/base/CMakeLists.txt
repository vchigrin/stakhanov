# Read version and add it as a preprocessor define.

    REPLACE "LASTCHANGE=" "" STAKHANOV_LASTCHANGE
    ${STAKHANOV_LASTCHANGE_LINE})
add_definitions(-DSTAKHANOV_LASTCHANGE=\"${STAKHANOV_LASTCHANGE}\")
